
```sql
SELECT R.WORKORDERID, R.TITLE, R.FIRST_NAME, SUM(R.HORA) AS HORA FROM (
       SELECT 
             (A.TIMESPENT/3600000.0) AS HORA,
             DATEADD(
                    S,
                    DATEDIFF(S,GETUTCDATE(),GETDATE()) + (A.TS_STARTTIME/1000) + 3600,
                    '1970-01-01 00:00:00'
             ) AS FECHA_INICIO,
             C.WORKORDERID,
             C.TITLE,
             D.FIRST_NAME
       FROM ChargesTable A
       INNER JOIN WorkOrderToCharge B ON B.CHARGEID=A.CHARGEID
       INNER JOIN WorkOrder C ON C.WORKORDERID = B.WORKORDERID
       INNER JOIN AaaUser D ON D.USER_ID=A.TECHNICIANID
       WHERE D.USER_ID IN(5,6,9,10,12,13,2406)
) R WHERE CAST(R.FECHA_INICIO AS DATE) BETWEEN '2024-07-01' AND '2024-07-31'
GROUP BY R.WORKORDERID, R.TITLE, R.FIRST_NAME
ORDER BY R.FIRST_NAME
```
