```sql
SELECT 
E.TASKTYPENAME AS TIPO,
A.TITLE AS ACTIVIDAD,
C.FIRST_NAME AS DESARROLLADOR,
CAST(DATEADD(S,DATEDIFF(S,GETUTCDATE(),GETDATE()) + (A.SCHEDULEDSTARTTIME/1000),'1970-01-01 00:00:00') AS DATE) AS FECHA_INICIO,
CAST(DATEADD(S,DATEDIFF(S,GETUTCDATE(),GETDATE()) + (A.SCHEDULEDENDTIME/1000),'1970-01-01 00:00:00') AS DATE) AS FECHA_FIN,
D.STATUSNAME AS ESTATUS,
A.ESTIMATEDEFFORTHOURS AS HORA,
A.ADDTIONAL_COST AS IMPORTE,
A.PER_OF_COMPLETION AS PORC_AVANCE
FROM TaskDetails A
INNER JOIN WorkOrderToTaskDetails B ON B.TASKID=A.TASKID
INNER JOIN AaaUser C ON C.USER_ID=A.OWNERID
INNER JOIN StatusDefinition D ON D.STATUSID=A.STATUSID
INNER JOIN TaskTypeDefinition E ON E.TASKTYPEID=A.TASKTYPEID
WHERE B.WORKORDERID=1954 ORDER BY A.TASKINDEX
```
